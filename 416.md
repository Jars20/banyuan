# 生成一个大小为SIZE的随机数列
```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 8

int* swap(int*a,int i,int k){
    if(k!=i){
     int temp =a[i];
         a[i]=a[k];
         a[k]=temp;
    }
    return a;
}

int main(void) {
    int a[]={0,1,2,3,4,5,6,7};
     for (int i=0;i<SIZE-1;i++){
         int k;
         srand((unsigned)time(NULL));
         k = rand() % (SIZE-1-i) + i;
         swap(a, i, k);
     }
    for(int i=0;i<SIZE;i++){
        printf("%4d",a[i]);
    }
    printf("\n");
    
    
    return 0;
}
```
